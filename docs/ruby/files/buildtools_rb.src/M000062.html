def windows
        #$mwindows||=nil
        return false if $xcompile
        if $mwindows==nil
                $mwindows||=(have_library("opengl32"))
                if not $mwindows
                        $mwindows=false
                        puts "no opengl32 found - so expecting _not_ to use windows"
                end
        end
        return $mwindows
end
