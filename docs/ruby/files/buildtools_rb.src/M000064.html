def test_program(p,er)
        path=ENV["PATH"]
        mp=""
        path.split(":").each{|s|
                c=s+"/"+p
                if File.exists?(c)
                        mp=c
                end
        }
        if mp==""
                puts "NOTE: "+er
        else
                puts "Found #{p} at #{mp}"
        end
        return mp
end
