def continueCampaign(campaign)
        begin
                $campaign=campaign
                part=campaign.getCurrentPart
                part.play
                if part.result.won=="won"
                        if campaign.proceed
                                break
                        end
                end
        end while part.result.won!="canceled"
        campaign.save("campaign0.antcmp") # FIXME:test save
        $campaign=nil
        puts "EXIT CAMPAIGNING"
end
