def addForest(map)
        tt=FOREST
        thres=1.0
        (0..(map.getW-1)).each{|x|
                (0..(map.getH-1)).each{|y|
                        if map.get(x,y)>thres
                                r=rand
                                map.setTerrain(x,y,tt,r)
                                if r>0.97 and map.getTerrain(x,y,ROCK)<0.3
                                        # add tree
                                        t=AntTree.new
                                        t.setPos(AGVector2.new(x,y))
                                        map.insertEntity(t)
                                end
                        end
                }
        }
end
