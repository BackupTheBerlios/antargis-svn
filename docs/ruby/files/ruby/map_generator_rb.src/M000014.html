def addGrass(map)
        tt=GRASS
        thres=0.3
        (0..(map.getW-1)).each{|x|
                (0..(map.getH-1)).each{|y|
                        if map.get(x,y)>thres
                                map.setTerrain(x,y,tt,rand)
                        end
                }
        }
        
        tt=GRASS2
        thres=1.3
        (0..(map.getW-1)).each{|x|
                (0..(map.getH-1)).each{|y|
                        if map.get(x,y)>thres
#                               map.setTerrain(x,y,tt,rand)
                                r=rand
                                map.setTerrain(x,y,tt,r)
                                if r>0.97
                                        # add tree
                                        t=AntGrass.new
                                        t.setPos(AGVector2.new(x,y))
                                        map.insertEntity(t)
                                end
                        end
                }
        }
end
