def generateMap(map,params)
        w=params.size
        h=params.size

        map.newMap(w,h)
        getMap.setHeight(2) # everything deep water

                flat=[[30,10],[15,20],[10,8],[5,40],[4,10],[3,20],[2,20],[1,70]]
                flat.each{|add|
                        for i in 1..add[1]
                                rad=add[0]
                                x=rand*(w-2*rad)+rad
                                y=rand*(h-2*rad)+rad
                                #map.addFlat(x,y,30,rad)
                                addFlatOnMap(map,x,y,2,rad)
                        end
                }

        blurMap(map)
        blurMap(map)
        blurMap(map)
        
        checkBeachSteepness(map)

        addWaterTerrain(map)
        blurTerrain(map,WATER)

        addGrass(map)
        addRock(map)
        addForest(map)

        map.mapChanged
end
