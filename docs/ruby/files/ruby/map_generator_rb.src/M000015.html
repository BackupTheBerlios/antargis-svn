def addRock(map)
        w=map.getW
        h=map.getH
        flat=[[5,10],[4,10],[3,20],[2,20]]
        flat.each{|add|
                for i in 1..add[1]
                        rad=add[0]
                        x=rand*(w-2*rad)+rad
                        y=rand*(h-2*rad)+rad
                        #map.addFlat(x,y,30,rad)
                        if map.get(x.to_i,y.to_i)>1
                                addFlatOnMapTerrain(map,x,y,1,rad,ROCK)
                        end
                end
        }
end
