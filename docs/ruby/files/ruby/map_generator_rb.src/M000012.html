def addWaterTerrain(map)
        tt=WATER
        (0..(map.getW-1)).each{|x|
                (0..(map.getH-1)).each{|y|
                        if map.get(x,y)<0
                                map.setTerrain(x,y,tt,0.7)
                        end
                }
        }
end
