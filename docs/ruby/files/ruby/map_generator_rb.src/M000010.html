def blurMap(map)
        vps=[[0,0],[1,0],[0,1],[1,1]]
        (0..(map.getW-2)).each{|x|
                (0..(map.getH-2)).each{|y|
                        c=0
                        vps.each{|v|
                                c+=map.get(x+v[0],y+v[1])
                        }
                        c/=vps.length
                        map.set(x,y,c)
                }
        }
end
