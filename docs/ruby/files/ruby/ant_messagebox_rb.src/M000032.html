def messageBox(title,text,flags,&block)
        mw=getApplication.getMainWidget
        w=MessageBox.new(mw,"data/gui/layout/messagebox.xml")
        w.block=block
        w.getChild("title").setText(title)
        w.getChild("text").setText(text)
        mw.addChild(w)
end
