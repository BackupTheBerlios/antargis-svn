def addMessageTypes
        desc=getDescendantsOfClass(Message).sort{|a,b|a.to_s<=>b.to_s}
        $messageType2ID={}
        $messageID2Type={}
        (0..(desc.length-1)).each{|id|
                d=desc[id]
                $messageType2ID[d]=id
                $messageID2Type[id]=d
                #puts "INIT:#{$messageID2Type[id]} #{id}"
        }
end
