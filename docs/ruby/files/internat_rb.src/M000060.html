def checkNode(node)
        res=[]
        node.getChildren.each{|c|
                res+=checkNode(c)
        }
        caption=node.get("caption")
        if caption!=""
                res.push(caption)
        end
        caption=node.get("text")
        if caption!=""
                res.push(caption)
        end
        res.collect{|s|s.gsub("\n","\\n")}
end
