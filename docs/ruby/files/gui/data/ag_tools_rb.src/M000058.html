def addSignal(name)
        eval "class \#{self.class}\n        def \#{name}(e=nil)\n                if e\n                        e.setCaller(self)\n                        return @\#{name}.signal(e)\n                else\n                        return @\#{name}\n                end\n        end\nend\n"
        eval("@#{name}=AGSignal.new(self,'#{name}')\n")
end
