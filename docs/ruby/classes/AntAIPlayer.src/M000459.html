        def move(time)
                if @heronames.length then
                        @heronames.each{|n|
                                ent=getMap.getByName(n)
                                if ent.is_a?(AntBoss)
                                        ent.setPlayer(self)
                                        @heroes.push(ent)
                                else
                                        puts "HERO NOT FOUND: '#{n}'"
                                end
                        }
                        @heronames.clear
                end
                @ai.eventFrame(time)
        end
