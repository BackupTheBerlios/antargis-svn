        def eventModBar(e)
                name=e.getCaller.getName
                what=name[0..2]
                barname=name[3..-1].downcase
                case what
                        when "inc"
                                @bars[barname].setValue(@bars[barname].getValue+0.1)
                        when "dec"
                                @bars[barname].setValue(@bars[barname].getValue-0.1)
                end
                updateVolumesP
        end
