        def assignJob(e)
                if @job.nil?
                        if e.getMode=="makeField"
                                pos=e.getPos2D
                                f=AntField.new
                                f.setPos(pos)
                                getMap.insertEntity(f)
                                puts "INSERT ENT #{@fields.length}"
                                #getMap.endChange
                                @fields.push(f)
                                f.farm=self
                                e.setMode("")
                                e.newMoveJob(0,getPos2D,0)
                                return
                        elsif @fields.length<7
                                e.setMode("makeField")
                                angle=getRand*360
                                radius=getRand*3+6
                                e.newMoveJob(0,getPos2D+AGVector2.new(AGAngle.new(angle))*radius,0)
                                return
                        end
                end
                super
        end
