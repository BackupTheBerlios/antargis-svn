        def eventResChange(e)
                case e.getCaller.getName
                        when "640"
                                setRes(640,480)
                        when "800"
                                setRes(800,600)
                        when "1024"
                                setRes(1024,768)
                        when "1280"
                                setRes(1280,1024)
                        when "1400"
                                setRes(1400,1050)
                end
                return true
        end
