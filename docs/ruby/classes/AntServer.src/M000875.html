        def eventFrame(t)
                delay(20) # server should run at relatively low FPS=20 - this should be sufficient and saves CPU power
                @map.move(t)

                @server.processMessages(@map)
                @framenr+=1
                if @framenr>50
                        @server.sendMessageToAll(TimeMessage.new(getMap.getTime))
                        @framenr=0
                end
        end
