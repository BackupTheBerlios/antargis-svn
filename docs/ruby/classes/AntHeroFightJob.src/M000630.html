        def check(man)
                # check state
                if @state=="fighting"
                        if getMenState("defeated").length==@men.length
                          lost
                        end
                        if not @target.getJob.is_a?(AntHeroFightJob) # enemy gave up
                                @finished=true
                        end
                end

                return if @finished

                puts "CHECK(FIGHT) #{man} #{man.getName}"
                case @state
                        when "fighting"
                                if [AntMan,AntHero].member?(man.class)
                                        startFightingMan(man)
                                end
                        else
                                super
                end
        end
