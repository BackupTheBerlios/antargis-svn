        def lost
                if not @finished
                        dputs "LOST!!!!!!!!!!"
                        @finished=true

                        if @hero.is_a?(AntHero)
                                @hero.eventHLJobFinished(self)
                                @hero.newHLMoveJob(0,@sitpos[@hero],0)
                        end

                        if @target.getJob.is_a?(AntHeroFightJob)
                                @target.getJob.won
                        end
                        @hero.setOwner(@target)
                        @hero.killAllJobs
                        @target.killAllJobs
                end
        end
