        def startFighting
                puts "--------------------"
                puts "START FIGHTING #{@hero}"
                if @state=="fighting"
                        raise "already fighting"
                end
                @state="fighting"
                @target.eventGotHLFight(@hero)
                @target.eventAttacked(@hero)
                initSitpositions

                if @fightMap.nil?
                        computeFightMap
                end
                
                @hero.getMen.each{|man|
                        puts man
                        man.setMode("fight")
                        startFightingMan(man)
                }
                puts "--------------------"
                @fightStarted=true
        end
