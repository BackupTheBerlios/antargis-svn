        def computeFightMap
                @fightMap={}
                puts "mymen: #{@hero.getMen.join(" ")}"
                puts "otheren: #{@target.getMen.join(" ")}"

                if @target.getJob.is_a?(AntHeroFightJob)
                        map=@target.getJob.fightMap
                        if map
                                map.each{|k,v|
                                        @fightMap[v]=k
                                }
                        end
                end
                undefeatedMen.each{|m|
                        if @fightMap[m].nil?
                                # find next enemy
                                e=findNextEnemy(m,@fightMap.values)

                                @fightMap[m]=e
                        end
                }
                @fightMap.each{|m,e|
                        puts "FIGHTMAP:#{m}:#{e}"
                }
        end
