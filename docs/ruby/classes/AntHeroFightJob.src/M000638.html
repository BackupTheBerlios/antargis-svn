        def haveDefeated(e)
                puts "haveDefeated: #{e.getName}"
                @men.each{|m|
                        puts "checking: #{m.fightTarget} #{e}"
                        if m.fightTarget==e
                                puts "#{m.getName} not longer attacks #{e.getName}"
                                m.delJob # search other target
                        end
                }
                @fightMap.delete_if{|k,v|v==e}
        end
