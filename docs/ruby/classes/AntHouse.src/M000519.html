        def getDescription
                m="man"
                home=_("{1} of them are at home.",@atHome.uniq.length)
                if getMen.length>1
                        m="men"
                end

                r=_("This is a {1}. It was build {2} years ago. Currently {3} {4} live here.",_(houseType),age,getMen.length,_(m))+home

                morale=0
                @men.each{|m|morale+=m.getMorale}
                if @men.length>0
                        morale/=@men.length
                end
                if morale>0.8
                        s=_("in good mood.")
                elsif morale>0.5
                        s=_("content.")
                elsif morale>0.2
                        s=_("discontent.")
                else
                        s=_("bad tempered.")
                end
                r+=_("They are {1}",_(s))
                r
        end
