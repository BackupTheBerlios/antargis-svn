        def getTargetPos
                p=getPos2D
                tries=10
                # assure that sheep doesn't walk into water
                while tries>0 do
                        d=AGVector2.new(rand-0.5,rand-0.5).normalized*2
                        t=p+d
                        t=getMap.truncPos(t)
                        tries-=1
                        if getMap.getHeight(t.x,t.y)<0
                                return t
                        end
                end 
                return p
        end
