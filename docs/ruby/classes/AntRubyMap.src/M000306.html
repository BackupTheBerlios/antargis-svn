        def getOwnHeroes()
                ents=getAllEntitiesV
                ret=[]
                ents.each{|eptr|
                        ent=eptr.get
                        #dputs ent
                        if ent.class==AntHero
                                p=ent.getPlayer
                                #dputs "player:",p,p.class
                                if p
                                        if p.class==AntHumanPlayer
                                                ret.push(ent)
                                        end
                                end
                        end
                }
                return ret
        end
