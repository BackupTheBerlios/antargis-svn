        def move(time)
                if @pause
                        return
                end
                time*=1 # increase speed
                @curTime+=time
                @@systemTime+=time
                super(time)
                
                checkTriggers
                
                @players.each{|player|
                        player.move(time)
                }
                
                AntSound.ambientSound(time)
                if not @started
                        @started=true
                        if @script
                                @script.eventLevelStarted
                        end
                end
        end
