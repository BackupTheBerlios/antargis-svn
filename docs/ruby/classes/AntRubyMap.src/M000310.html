        def saveXML(n)
                super(n)
                @players.each{|player|
                        c=n.addChild(player.xmlName)
                        player.saveXML(c)
                }
                @targets.each{|name,t|
                        c=n.addChild(t.xmlName)
                        t.saveXML(c)
                }
                @triggers.each{|t|
                        c=n.addChild(t.xmlName)
                        t.saveXML(c)
                }
                if @scriptClass
                        n.set("scriptclass",@scriptClass)
                end
                if @scriptFile
                        n.set("scriptfile",@scriptFile)
                end
                if @script
                        c=n.addChild("scriptdata")
                        @script.saveXML(c)
                end
        end
