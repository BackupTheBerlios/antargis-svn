        def checkPackMerge
                # find next leader
                setProvide("wolf_leader",false)

                tent=getMap.getNext(self,"wolf_leader")
                if tent
                        if (tent.getPos2D-getPos2D).length<10
                                if not tent.inPack(self)
                                        # simple take me as leader
                                        tent.setLeader(self)
                                end
                        end
                end
        end
