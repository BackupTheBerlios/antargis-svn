        def getTargetPos
                p=getPos2D
                t=p
                ok=true
                radius=4
                # assure that sheep doesn't walk into water
                while getMap.getHeight(t.x,t.y)<0.1 or ok do
                        d=AGVector2.new(rand-0.5,rand-0.5).normalized*radius
                        t=p+d
                        t=getMap.truncPos(t)
                        ok=false
                end 
                return t
        end
