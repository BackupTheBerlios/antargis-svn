        def assignJob(wolf)
                case @job
                        when :resting
                                wolf.newRestJob(4)
                                wolf.setMode("resting")
                        when :killsheep
                                if @sheep
                                        puts "KILLSHEEP #{@sheep} #{self}"
                                        if (@sheep.getPos2D-wolf.getPos2D).length>0.5
                                                wolf.newMoveJob(0,@sheep.getPos2D,0)
                                        else
                                                @sheep.eventDie   
                                                if @leader
                                                        @leader.shareFood
                                                else
                                                        shareFood
                                                end
                                        end
                                end
                end
        end
