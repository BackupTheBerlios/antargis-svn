        def doRubber(list,button)
                puts "DOING RUBBER"
                list.each{|res|
                        mesh=res.node
                        if mesh.class==Mesh
                                ent=getMap.getEntity(mesh)
                                if ent
                                        getMap.removeEntity(ent)
                                else
                                        puts "NOT FOUND!"
                                end
                                break
                        end
                }
                getMap.move(0)
        end
