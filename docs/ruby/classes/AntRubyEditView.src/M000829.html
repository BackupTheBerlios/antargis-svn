        def addEntity(list,button)
                pos=nil
                list.each{|c|
                        if isTerrain(c.node)
                                pos=c.pos
                        end
                }
                if not pos
                        return
                end
                puts "ADDENTITY"
                #pos=getMarkerPos
                dorand=true
                if @type==AntDeco
                        tree=@type.new(@decoType)
                        if @decoType=="floor" or @decoType=="block"
                                dorand=false
                        end
                elsif @type==AntHero
                        tree=@type.new
                        tree.setAppearance(@appearance)
                else
                        puts "TYPE:",@type
                        tree=@type.new
                end
                puts "type:"
                puts tree
                tree.setPos(AGVector2.new(pos.x,pos.y))
                getMap.insertEntity(tree)
        end
