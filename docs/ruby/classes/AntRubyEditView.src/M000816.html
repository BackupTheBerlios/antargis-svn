        def setTab(name)
                ["terrain","entities"].each{|e|
                        if e==name
                                w=@buttonlayout.getChild(e+"Tab")
                                if w
                                        w.show
                                end
                                w=@buttonlayout.getChild(e)
                                w.setChecked(true)
                        else
                                w=@buttonlayout.getChild(e+"Tab")
                                if w
                                        w.hide
                                end
                                w=@buttonlayout.getChild(e)
                                w.setChecked(false)
                        end
                }
                @modifier=@modifiers[name]
        end
