        def inspect(e)
                getChild("shutter").hide
                if e.getName!="" and (not e.getName=~/0\./)
                        setTitle(e.getName)
                else
                        n=e.class.to_s
                        if n[0..2]=="Ant"
                                n.sub!("Ant","")
                        end
                        setTitle(n)
                end
                @inspect=e
                getChild("InvBar").setHero(e)
                updateInspection
        end
