        def AntSound.updateVolumes
                getSoundManager.volumeMusic(getVolume("music"))
                if @@normalVolume
                        setNormalVolumeWave
                else
                        setQuietVolumeWave
                end
                if @@ambientSound
                        getSoundManager.stopChannel(@@ambientSound)
                        @@ambientSound=nil
                end
                ambientSound(1)
        end
