        def AntSound.updateSoundPos(scene)
                @@loopSounds.each{|id,a|
                        pos=a[2]
                        d=((scene.getCamera.dim2-pos).length-INNER_VOL_SIZE)
                        vol=1
                        if d>0
                                vol=[(OUTER_VOL_SIZE-d)/OUTER_VOL_SIZE,0].max
                        end
                        getSoundManager.volumeSound(a[0],a[3]*vol)
                }
        end
