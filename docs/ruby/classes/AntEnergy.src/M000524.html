        def draw(p)
                super
                r=p.getRect.origin
                w=r.width
                h=r.height-@vborder

                v=@vborder/2

                mh=h/(@values.length*3)
                
                i=0
                @keys.each{|k|
                        v=check(@values[k])
                        if v<0
                                puts v
                        raise 1
                        end
                        c=@colors[k]
                        a0=c*0.7
                        a1=c*0.7
                        a2=c*1.0
                        a3=c*0.8
                
                        mr=AGRect2.new(@hborder,i*mh,(w-@hborder*2)*v,mh*2)

                        p.drawGradient(mr,a0,a1,a2,a3)
                        i+=3
                }
        end
