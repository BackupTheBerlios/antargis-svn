        def startWalking
                puts "startWalking:::::::::::"
                @hero.formation=AntFormationBlock.new(@hero,formatDir)
                @pos=@waypoints.shift
                @men.each{|m|
                        f=@hero.getFormation(m,@pos)
                        m.newMoveJob(0,f,@dist)
                        m.setMode("moving")
                }
                @state="moving"
                puts "startWalking."
        end
