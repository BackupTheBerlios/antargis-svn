        def startFormatting
                puts "startFormatting:::::::::::"
                @hero.formation=AntFormationBlock.new(@hero,formatDir)
                @men.each{|man|
                        f=@hero.getFormation(man,@hero.getPos2D)
                        man.newMoveJob(0,f,@formatDist)
                        man.setMode("format")
                }
                @state="format"
                puts "startFormatting."
        end
