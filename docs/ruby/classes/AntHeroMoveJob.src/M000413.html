        def check(man)
                if @moveFinished
                        return       
                end
                puts "CHECK #{@state} #{man.getMode} #{man}"
                case @state
                        when "format"
                                if man.getMode=="format"
                                        man.setMode("ready")
                                end
                                if getMenState("format").length==0
                                        startWalking
                                end
                        when "moving"
                                if @waypoints.length>0
                                        startFormatting
                                else
                                        self.moveFinished=true
                                end
                end
        end
