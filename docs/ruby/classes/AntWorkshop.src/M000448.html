        def assignJob(e)
                if e.getMode=="invent"
                        produceOneTool
                        e.setMode("any")
                        decSmoke
                end
                if not @job
                        if allThereForTool && notEnoughTools && atHome(e)
                                e.setMode("invent")
                                e.newRestJob(10)
                                startToolProduction
                                return
                        end
                end
                super(e)
        end
