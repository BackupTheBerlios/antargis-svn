        def build(place,type)
                return unless place.is_a?(AGVector2) and type.is_a?(String)
                if valid
                        ds=getDescendantsOfClass(AntHouse)
                        @map={}
                        ds.each{|d|
                                c=getChild(d.to_s)
                                if c
                                        addHandler(c,:sigClick,:eventBuild)
                                        @map[d.to_s]=d
                                end
                        }
                        type=@map[type]
                        if type
                                @ent.newHLBuildJob(place,type)
                        end
                end
        end
