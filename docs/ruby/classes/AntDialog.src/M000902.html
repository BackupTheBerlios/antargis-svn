        def eventKeyDown(e)
                if super then return true end
                case e.getKey
                        when SDLK_ESCAPE
                                eventClose(e)
                                getMap.pause=false
                                return true
                        when SDLK_RETURN, SDLK_SPACE
                                eventOk(e)
                end
                return false
        end
