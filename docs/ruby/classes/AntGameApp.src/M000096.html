        def showPanel
                sideBar=@layout.getChild("SideBar")
                sideBarRight=@layout.getChild("SideBarRight")
                sideBar.show
                @controls=true
                # reset Scene's width
                getScene.getCameraObject.setWidth((getScreen.getWidth-sideBarRight.width).to_i)
        end
