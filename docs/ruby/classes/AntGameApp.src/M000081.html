        def eventMapClicked(pos,button)
                puts "#{@job} #{button}"
                if @job and button==1 then
                        case @job
                                when "doBuild"
                                        buildHouse(pos.dim2)
                                        @job=nil
                        end
                        resetJob
                        return
                end
                if @hero and button==3 then
                        # assign hero a move job
                        @hero.newHLMoveJob(0,pos,0)
                end
        end
