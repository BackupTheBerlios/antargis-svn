        def tellStory(flow)
                getMap.pause=true
                hidePanel
                if @story.nil?
                        @story=AntStoryTalk.new(@layout)
                end
                @layout.addChild(@story)
                @story.show
                @story.setFlow(flow)
                addHandler(@story,:sigStoryFinished,:eventStoryTalkFinished)
        end
