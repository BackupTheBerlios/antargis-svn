        def checkHeroEnergy
                name=nil
                getMap.getPlayer.getHeroes.select{|h|
                        if h.class==AntHero
                                if h.getEnergy<0.3
                                        name=h.getName
                                end
                        end
                }
                if @infobox==nil and name!=nil
                        @layout.addChild(@infobox=AntInfoBox.new(@layout,_("Your hero {1} suffers.",name)))
                elsif @infobox!=nil and name==nil
                        @infobox.close
                        @infobox=nil
                end
        end
