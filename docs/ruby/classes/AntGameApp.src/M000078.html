        def eventHeroButton(e)
                c=e.getCaller.getName
                num=c[4..4]
                name=@layout.getChild("HeroName#{num}").getText
                ent=getMap.getByName(name)
                selectHero(ent)
                if ent==@inspect
                        focusHero(ent)
                end
                inspectEntity(ent)
                return true

        end
