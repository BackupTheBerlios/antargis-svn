        def checkOnWater(name)
                z=getMap.getPos(getPos2D).z

                        # under water
                        if name=~/sit/ and not isOnOpenWater
                                name="stand"

                        end
                        if isOnOpenWater and haveBoat
                                puts "onOpenWater"
