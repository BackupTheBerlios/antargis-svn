        def eventDie
                simDeath
                #eventDefeated
                newRestJob(20)
                setMeshState("dead")
                playSound("die")

                if @boss
                        @boss.removeMan(self)
                end
                if not self.resource.empty
                        sack=AntSack.new(getPos2D+AGVector2.new(0.3,-0.3))
                        getMap.insertEntity(sack)
                        sack.resource.takeAll(self.resource)
                        sack.resourceChanged
                end
        end
