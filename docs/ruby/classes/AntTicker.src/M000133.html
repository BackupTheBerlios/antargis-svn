        def draw(p)
                now=@menuapp.getTicks
                if @starttime>0
                        @y+=(now-@starttime)*@speed
                end
                @starttime=now
                ty=0
                
                @text.each{|r|
                        font=nil
                        if r=~/_/    
                                #r.sub!("_","")
                                r=r[1..(r.length-1)]
                                font=@cfont
                        else
                                font=@nfont
                        end

                        r=AGStringUtf8.new(r)

                        w=font.getWidth(r)
                        tx=((width-w)/2).to_i
                        p.renderText(r,AGVector2.new(tx,(ty-@y)),font)
                        ty+=40
                }
        end
