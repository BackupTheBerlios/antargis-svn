        def initialize(node)
                @screens=[]
                node.getChildren("screen").each{|s|
                        screen={}
                        screen[:text]=""
                        s.getChildren("image").each{|c|
                                imageName=c.get("filename")
                                screen[:imageName]=imageName
                                if imageName
                                        screen[:image]=AGSurface.load(imageName)
                                end
                        }
                        s.getChildren("text").each{|c|
                                screen[:text]+=c.get("text")
                        }
                        @screens.push(screen)
                }
                #@finished="won"
        end
