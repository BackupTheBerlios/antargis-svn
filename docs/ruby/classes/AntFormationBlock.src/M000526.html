        def calcFormation
                boss=@boss
                dir=@dir
                # precalc formation and store into @rpos
        
                men=boss.getMen
                # sort from back to front (lines)
                men=sort(men)

                vpos={}
                @rpos={}
                linesizes={}
                line=0
                row=0

                rowsize=5

                normal=dir.normal
                men.each{|m|
                        vpos[m]=[row,line]
                        
                        linesizes[line]||=0
                        linesizes[line]+=1
                        row+=1
                        if row>rowsize-1 # or weapon-group changed
                                row=0
                                line+=1
                        end
                }
                men.each{|m|
                        row,line=vpos[m]
                        linesize=linesizes[line]
                        @rpos[m]=normal*(row-(linesize/2.0))*0.5 + dir*line*0.7
                }


        end
