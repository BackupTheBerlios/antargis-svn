        def getDescription
                a=_(building.to_s.gsub("Ant",""))
                d=_("This will be a {1}.",a)
                if @steps
                        b=((@@buildingSiteMeshes.length-1)-@progress)*@steps/(@@buildingSiteMeshes.length-1)
                        b=b.to_i.to_s
                        d+=_("It will be ready after {1} more steps.",b)
                end
                d
        end
