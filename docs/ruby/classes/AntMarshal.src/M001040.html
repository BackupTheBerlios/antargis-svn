        def AntMarshal.dump(x)
                puts "try marshalling #{x}"
                d=Marshal.dump(x)
                len=d.length

                puts "ANtMarshal.dump            LEN:#{len}"

                puts d[0].to_s+" "+d[1].to_s+" "+d[2].to_s

                # add length (4 bytes) in network-byte-order
                [len].pack("N")+d
        end
