        def check(man)
                if man.is_a?(AntHero)
                        if @firstTime
                                man.newRestJob(@time)
                                @firstTime=false
                        end
                        eat(man)
                        #spreadThings
                        return
                end
                case man.getMode
                        when "rest_eat"
                                spreadThings
                                eat(man)
                                sit(man)
                                man.setMode("rest_sit")
                        when "rest_sit"
                                sit(man)
                                if man.getFood<0.5 and heroHasFood
                                        man.newMoveJob(0,@hero.getPos2D,0)
                                        man.setMode("rest_eat")
                                end
                        else
                                # rest_eat
                                man.newMoveJob(0,@hero.getPos2D,0)
                                man.setMode("rest_eat")
                end
        end
