        def playSound(name,minDiff=0.5)
                scene=getMap.getScene
                d=((scene.getCamera.dim2-getPos2D).length-INNER_VOL_SIZE)
                vol=1
                if d>0
                        vol=[(OUTER_VOL_SIZE-d)/OUTER_VOL_SIZE,0].max
                end
                AntSound.playSoundGlobal(name,vol,minDiff)
        end
