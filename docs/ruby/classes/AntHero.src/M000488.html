        def setFire(flag)
                puts "setFire #{flag}"
                if flag
                        if getPos3D.z>0 # won't start fire in water!!
                                if not @fire
                                        @fire=AntFire.new(getPos3D+AGVector3.new(0.7,0,0))
                                        getMap.insertEntity(@fire)
                                end
                                startFireSound
                        end
                else
                        #raise 1
                        if @fire
                                @fire.disable
                                @fire=false
                        end
                        stopFireSound
                end
        end
