        def moveHome(man)
                if @formation.nil?
                        raise "formation is nil!"
                end
                pos=@formation.getPosition(man)
#               pos=getSitFormation(man)
                if (man.getPos2D-pos).length>1
                        man.newMoveJob(0,pos,0)
                else
                        man.newRestJob(2)
                end
        end
