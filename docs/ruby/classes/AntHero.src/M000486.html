        def newHLDismissJob()
                @job=nil
                agg=getAggression
                men=@men.select{|m|not m.is_a?(AntHero)} # exclude hero
                c=(men.length-1)*agg/3
                        
                men=men[0..c]
                men.each{|m|
                        # seek new boss
                        b=getMap.getNextList(m,"house",0).collect{|e|e.get}.select{|e|e.getPlayer==getPlayer}.shuffle[0]
                        if b
                                m.setBoss(b)
                        else
                                m.setNoBoss
                        end
                        @men.delete(m)
                        m.delJob
                }
                getMap.eventHLDismissed(self)
        end
