        def check(man)
                if moveFinished
                        case @state
                                when "eat"
                                        man.newRestJob(10)
                                        @toEat-=1
                                        if @toEat<=0
                                                killAnimal
                                                playSound
                                                @finished=true
                                        end
                                else
                                        @state="eat"
                                        @men.each{|m|
                                                m.newMoveJob(0,@target.getPos2D,0)
                                        }
                                        @toEat=@men.length
                        end
                else
                        super
                end
        end
