        def processMessage(m)
                case m
                        when ChallengeMessage
                                sendMessage(LoginMessage.new(@name,myhash(m.challenge+myhash(@name+@pw)),getProtocolVersion))
                                puts "try login"
                                return true
                        when WelcomeMessage
                                @state=:loggedin
                                super
                                return true
                        else
                                super
                end
                return false
        end
