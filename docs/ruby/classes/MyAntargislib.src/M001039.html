        def MyAntargislib.startGC
                if not @@antProfiling
                        puts "starting GC"
                        GC.enable
                        GC.start
                        GC.disable
                        puts "GC ok"
                else
                        puts "starting GC"
                        GC.enable
                        GC.disable
                        puts "GC ok"
                end
        end
