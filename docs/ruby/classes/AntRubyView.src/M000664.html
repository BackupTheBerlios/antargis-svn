        def eventHover(list,button)
                if list.length>0
                        n=list[0]
                        mesh=n.node
                        ok=false
                        if mesh.class==Mesh
                                ent=getMap.getEntity(mesh)
                                if ent
                                        if ent.is_a?(AntBoss) then
                                                hoverEntity(ent)
                                                ok=true
                                        end
                                end
                        end
                        if not ok
                                hoverEntity(nil)
                        end
                end
                return true
        end
