        def updateNamePositions
                heroes=@map.getHeroes
                @names.each{|k,w|
                        found=false
                        heroes.each{|hero|
                                if hero.getName==w.getText
                                        w.setRect(getHeroScreenPos(hero))
                                        found=true
                                end
                        }
                        if (not found)
                                w.hide
                        end
                }
        end
