        def initialize(parent)
                super(parent,"data/gui/layout/loaddialog.xml")
                d=getDirectory(getSavePath)
                @lb=toAGListBox(getChild("Files"))
                puts d
                d.each { |x|
                        if x =~ /.*\.antlvl/ then
                                puts "FILE:"+x
                                value=x.gsub(".antlvl","")
                                @lb.insertItem(x,value)
                        end
                }
        end
