        def Rookey.link_dll(name,files)
    files.flatten!
    files.uniq!
          config=Rookey::getConfig
          
    linker=Rookey::Compiler.new(config)
    
    target=linker.dllName(name)
    
    desc "Link DLL #{name}"
    task target=>files do |t|
      linker.linkDLL(t)
    end
    CLEAN << target
    
    target
        end
