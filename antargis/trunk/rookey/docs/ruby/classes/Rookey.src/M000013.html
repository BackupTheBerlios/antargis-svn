        def Rookey.swigDirect(output,interface,config=nil)
          config||=Rookey::getConfig
          swigCompiler=Rookey::Swig.new(config)
          
          
          # FIXME: handle deps
          
          rule output=>Swig.getDeps(interface) do |t|

          end

        end
