  def Rookey.getConfig
    if @@config.nil?
      if File.exists?(Rookey::CONFIG_FILENAME)
        load Rookey::CONFIG_FILENAME
            else 
              runConfigure
        hibe(@@config,Rookey::CONFIG_FILENAME)
            end
      CLEAN << Rookey::CONFIG_FILENAME
    end
    @@config
  end
