  def Rookey.runConfigure
    configurators=getDescendantsOfClass(Configurator)
    ok=[]
    run=[]
    config=Config.new
    while run.length<configurators.length
            configurators.each{|c|
              unless run.member?(c)
                      if c.needs.select{|s|not ok.member?(s)}.length == 0
                        log "Running congurator #{c}"
                        c.new.run(config)
                        ok+=c.provides
                        ok.uniq!
                        run << c
                      end
              end
            }
    end
    puts
    @@config=config
  end
