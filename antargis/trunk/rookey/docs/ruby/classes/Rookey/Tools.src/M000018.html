                def Tools.mkdir(dir)
                  return if File.exists?(dir)
                  pp dir
                  begin
                    puts "making #{dir}"
                    Dir.mkdir(dir.to_s)
                    puts "ok"
                  rescue
                    p=File.split(dir)[0]
                    return if p==dir
                    Rookey.mkdir(p)
                    begin
                      Dir.mkdir(dir)
                    rescue
                    end
                  end
                end
