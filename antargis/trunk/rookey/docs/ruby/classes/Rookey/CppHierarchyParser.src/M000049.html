    def getAllRubyClasses
      clist=["AGRubyObject"]
      while true
        oldLength=clist.length
        @cppClasses.each{|c|
          if clist.member?(c.parent)
            clist=(clist+[c]).uniq
          end
        }
        break unless oldLength!=clist.length
      end
      clist
    end
